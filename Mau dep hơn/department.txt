  <script>
    const API_URL = 'department.api.php';

    function loadDepartments() {
      fetch(`${API_URL}?action=list`)
        .then(res => res.json())
        .then(result => {
          const data = result.data;
          const tableBody = document.querySelector('#departmentList');
          tableBody.innerHTML = '';

          if (data.length === 0) {
              tableBody.innerHTML = `<tr><td colspan="4" style="text-align: center;">Chưa có phòng ban nào.</td></tr>`;
              return;
          }

          data.forEach(row => {
            tableBody.innerHTML += `
              <tr>
                <td>${row.id_phongban}</td>
                <td>${row.tenphongban}</td>
                <td>${row.mota ?? ''}</td>
                <td>
                  <button class="btn btn-danger" onclick="deleteDepartment(${row.id_phongban})">Xoá</button>
                </td>
              </tr>
            `;
          });
        })
        .catch(error => {
          console.error("Lỗi khi tải phòng ban:", error);
          document.querySelector('#departmentList').innerHTML = `<tr><td colspan="4" style="text-align: center; color: red;">Không thể tải dữ liệu.</td></tr>`;
        });
    }

    function deleteDepartment(id) {
      if (!confirm("Bạn có chắc muốn xoá phòng ban này?")) return;

      fetch(`${API_URL}?action=delete`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ id_phongban: id })
      })
      .then(res => res.json())
      .then(res => {
        alert(res.message || "Đã xoá phòng ban.");
        loadDepartments();
      })
      .catch(error => {
          alert("Lỗi! Không thể xoá phòng ban. Kiểm tra console.");
          console.error("Lỗi khi xoá:", error);
      });
    }

    document.getElementById('addForm').addEventListener('submit', function(e) {
      e.preventDefault();
      const formData = new FormData(this);
      const data = Object.fromEntries(formData.entries());

      fetch(`${API_URL}?action=add`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data)
      })
      .then(res => res.json())
      .then(res => {
        alert(res.message || "Thêm phòng ban thành công!");
        this.reset();
        loadDepartments();
      })
      .catch(error => {
          alert("Lỗi! Không thể thêm phòng ban. Kiểm tra console.");
          console.error("Lỗi khi thêm:", error);
      });
    });

    window.onload = loadDepartments;
  </script>
