// Hàm tính thuế TNCN (Giữ nguyên)
function tinhThue($thuNhap) {
    if ($thuNhap <= 5000000) return $thuNhap * 0.05;
    elseif ($thuNhap <= 10000000) return $thuNhap * 0.10;
    elseif ($thuNhap <= 18000000) return $thuNhap * 0.15;
    elseif ($thuNhap <= 32000000) return $thuNhap * 0.20;
    elseif ($thuNhap <= 52000000) return $thuNhap * 0.25;
    elseif ($thuNhap <= 80000000) return $thuNhap * 0.30;
    else return $thuNhap * 0.35;
}

if ($view === 'pay') {
    // --- TỐI ƯU: 1 "SIÊU TRUY VẤN" THAY VÌ N+1 TRUY VẤN ---
    $query = "
        SELECT 
            nv.id_nhanvien, 
            nv.hoten, 
            cv.tenchucvu, 
            cv.hesoluong,
            COALESCE(cc.tong_gio, 0) AS tong_gio,
            COALESCE(pc_sum.tongphucap, 0) AS tongphucap,
            COALESCE(tp.tongthuong, 0) AS tongthuong,
            COALESCE(tp.tongphat, 0) AS tongphat
        FROM nhan_viens nv
        JOIN chuc_vus cv ON nv.id_chucvu = cv.id_chucvu
        LEFT JOIN (
            -- Tính tổng giờ làm
            SELECT 
                id_nhanvien, 
                ROUND(SUM(TIME_TO_SEC(TIMEDIFF(giora, giovao)) / 3600), 2) as tong_gio 
            FROM cham_congs 
            WHERE trangthai = 1 AND giovao IS NOT NULL AND giora IS NOT NULL
            GROUP BY id_nhanvien
        ) AS cc ON nv.id_nhanvien = cc.id_nhanvien
        LEFT JOIN (
            -- Tính tổng phụ cấp
            SELECT nvp.id_nhanvien, SUM(pc.sotien) AS tongphucap 
            FROM nhan_vien_phu_caps nvp 
            JOIN phu_caps pc ON nvp.id_phucap = pc.id_phucap 
            GROUP BY nvp.id_nhanvien
        ) AS pc_sum ON nv.id_nhanvien = pc_sum.id_nhanvien
        LEFT JOIN (
            -- Tính tổng thưởng/phạt
            SELECT 
                id_nhanvien, 
                SUM(CASE WHEN loai = 'thưởng' THEN sotien ELSE 0 END) AS tongthuong, 
                SUM(CASE WHEN loai = 'phạt' THEN sotien ELSE 0 END) AS tongphat 
            FROM thuong_phats 
            GROUP BY id_nhanvien
        ) AS tp ON nv.id_nhanvien = tp.id_nhanvien
        WHERE nv.trangthai = 1 -- Chỉ tính lương cho nhân viên đang làm
    ";
    $result_pay = $conn->query($query);
}
