  <script>
  document.addEventListener('DOMContentLoaded', () => {
    loadEmployees();
    loadOptions();

    document.getElementById('employeeForm').addEventListener('submit', async function(e) {
      e.preventDefault();
      const formData = new FormData(this);
      const data = Object.fromEntries(formData.entries());

      const res = await fetch('employee.api.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action: 'create', ...data })
      });
      const result = await res.json();
      alert(result.message || 'Thêm thành công');
      this.reset();
      loadEmployees();
    });

    async function loadEmployees() {
      const res = await fetch('employee.api.php?action=read');
      const data = await res.json();
      const tbody = document.querySelector('#employeeTable tbody');
      tbody.innerHTML = '';
      data.forEach(emp => {
        tbody.innerHTML += `
          <tr>
            <td>${emp.id_nhanvien}</td>
            <td>${emp.hoten}</td>
            <td>${emp.ngaysinh}</td>
            <td>${emp.gioitinh}</td>
            <td>${emp.email}</td>
            <td>${emp.sdt}</td>
            <td>${emp.diachi}</td>
            <td>${emp.ngayvaolam}</td>
            <td>${emp.tenphongban}</td>
            <td>${emp.tenchucvu}</td>
            <td>${emp.trangthai == 1 ? 'Đang làm' : 'Nghỉ việc'}</td>
            
            <td><button class="btn btn-danger" onclick="deleteEmployee(${emp.id_nhanvien})">Xoá</button></td>
          
          </tr>
        `;
      });
    }

    async function loadOptions() {
      const resPB = await fetch('get_phongban.php');
      const pb = await resPB.json();
      const pbSelect = document.getElementById('phongbanSelect');
      pb.forEach(p => {
        pbSelect.innerHTML += `<option value="${p.id_phongban}">${p.tenphongban}</option>`;
      });

      const resCV = await fetch('get_chucvu.php');
      const cv = await resCV.json();
      const cvSelect = document.getElementById('chucvuSelect');
      cv.forEach(c => {
        cvSelect.innerHTML += `<option value="${c.id_chucvu}">${c.tenchucvu}</option>`;
      });
    }
  });

  async function deleteEmployee(id) {
    if (confirm('Bạn có chắc muốn xoá nhân viên này?')) {
      const res = await fetch('employee.api.php', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action: 'delete', id_nhanvien: id })
      });
      const result = await res.json();
      alert(result.message || 'Đã xoá');
      location.reload(); // Tải lại trang để cập nhật (hoặc gọi lại loadEmployees())
    }
  }
