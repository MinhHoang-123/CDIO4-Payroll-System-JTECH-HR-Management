if (isset($_GET['delete'])) {
    $id_nv = $_GET['id_nhanvien'];
    $id_pc = $_GET['id_phucap'];
    
    $stmt_del = $conn->prepare("DELETE FROM nhan_vien_phu_caps WHERE id_nhanvien = ? AND id_phucap = ?");
    $stmt_del->bind_param("ii", $id_nv, $id_pc);
    $stmt_del->execute();
    $stmt_del->close();
    header("Location: subsidy.php?msg=delete_success");
    exit;
}

// Thêm loại phụ cấp mới
if (isset($_POST['add_phucap'])) {
    $tenphucap = $_POST['tenphucap'];
    $sotien = $_POST['sotien'];
    $ghichu = $_POST['ghichu'];
    $created_at = date("Y-m-d H:i:s");

    $stmt = $conn->prepare("INSERT INTO phu_caps (tenphucap, sotien, ghichu, created_at) VALUES (?, ?, ?, ?)");
    $stmt->bind_param("siss", $tenphucap, $sotien, $ghichu, $created_at);
    $stmt->execute();
    $stmt->close();
    header("Location: subsidy.php?msg=add_type_success");
    exit;
}

// Gán phụ cấp cho nhân viên (AN TOÀN & TỐT HƠN)
if (isset($_POST['add_subsidy'])) {
    $id_nhanvien = $_POST['id_nhanvien'];
    $id_phucap = $_POST['id_phucap'];
    $ngaybatdau = $_POST['ngaybatdau'];
    $ngayketthuc = $_POST['ngayketthuc'];

    // KIỂM TRA (AN TOÀN HƠN)
    $stmt_check = $conn->prepare("SELECT * FROM nhan_vien_phu_caps WHERE id_nhanvien=? AND id_phucap=?");
    $stmt_check->bind_param("ii", $id_nhanvien, $id_phucap);
    $stmt_check->execute();
    $check_result = $stmt_check->get_result();
    $stmt_check->close();

    if ($check_result->num_rows == 0) {
        $stmt_insert = $conn->prepare("INSERT INTO nhan_vien_phu_caps (id_nhanvien, id_phucap, ngaybatdau, ngayketthuc) VALUES (?, ?, ?, ?)");
        $stmt_insert->bind_param("iiss", $id_nhanvien, $id_phucap, $ngaybatdau, $ngayketthuc);
        $stmt_insert->execute();
        $stmt_insert->close();
        header("Location: subsidy.php?msg=assign_success");
        exit;
    } else {
        // Đã tồn tại, báo lỗi
        header("Location: subsidy.php?msg=assign_error");
        exit;
    }
}
