  <script>
    window.onload = loadRoles;

    function loadRoles() {
      fetch('role.api.php')
        .then(res => res.json())
        .then(data => {
          const table = document.getElementById('roleTable');
          table.innerHTML = '';
          if (data.status === 'success') {
            if (data.data.length === 0) {
                table.innerHTML = `<tr><td colspan="4" style="text-align: center;">Chưa có chức vụ nào.</td></tr>`;
                return;
            }
            data.data.forEach(role => {
              table.innerHTML += `
                <tr>
                  <td>${role.id_chucvu}</td>
                  <td>${role.tenchucvu}</td>
                  <td>${role.hesoluong}</td>
                  <td>
                    <button onclick='editRole(${JSON.stringify(role)})' class="btn btn-warning">Sửa</button>
                    <button onclick="deleteRole(${role.id_chucvu})" class="btn btn-danger">Xoá</button>
                  </td>
                </tr>
              `;
            });
          } else {
            table.innerHTML = `<tr><td colspan="4" style="text-align: center; color: red;">${data.message}</td></tr>`;
          }
        });
    }

    // HÀM MỚI: Dùng để xoá trắng form
    function resetForm() {
        document.getElementById('id_chucvu').value = '';
        document.getElementById('tenchucvu').value = '';
        document.getElementById('hesoluong').value = '';
        document.getElementById('tenchucvu').focus(); // Di chuyển con trỏ về ô đầu tiên
    }

    function saveRole() {
      const id = document.getElementById('id_chucvu').value;
      const ten = document.getElementById('tenchucvu').value.trim();
      const heso = document.getElementById('hesoluong').value;

      if (!ten || !heso) {
        alert("Vui lòng nhập đầy đủ thông tin");
        return;
      }

      const method = id ? 'PUT' : 'POST';
      const payload = {
        id_chucvu: id, // Gửi ID (dù là rỗng hay có)
        tenchucvu: ten,
        hesoluong: parseFloat(heso)
      };

      fetch('role.api.php', {
        method,
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      })
      .then(res => res.json())
      .then(data => {
        alert(data.message);
        if (data.status === 'success') {
          resetForm(); // Dùng hàm mới
          loadRoles();
        }
      });
    }

    function editRole(role) {
      document.getElementById('id_chucvu').value = role.id_chucvu;
      document.getElementById('tenchucvu').value = role.tenchucvu;
      document.getElementById('hesoluong').value = role.hesoluong;
      window.scrollTo(0, 0); // Cuộn lên đầu trang để thấy form
      document.getElementById('tenchucvu').focus();
    }

    function deleteRole(id) {
      if (!confirm("Bạn chắc chắn muốn xoá chức vụ này? (Nhân viên thuộc chức vụ này có thể bị ảnh hưởng).")) return;

      fetch('role.api.N-T-G-I.php', { // Tên file API của bạn
        method: 'DELETE',
        headers: { 'Content-Type': 'application/json' }, // Gửi JSON
        body: JSON.stringify({ id_chucvu: id }) // Gửi ID trong body
      })
      .then(res => res.json())
      .then(data => {
        alert(data.message);
        if (data.status === 'success') loadRoles();
      })
      .catch(error => {
          alert("Lỗi! Không thể xoá. Kiểm tra console.");
          console.error("Lỗi khi xoá:", error);
      });
    }

    // SỬA LOGIC XOÁ:
